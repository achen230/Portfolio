Google News Sentiment Analysis

Apple Stock News Sentiment Analyzer
A Python script that fetches recent news articles about Apple stock from Google News, performs sentiment analysis on the article descriptions using NLTK's VADER, and visualizes the average sentiment per day.

from gnews import GNews
import nltk
from nltk.sentiment import SentimentIntensityAnalyzer
import matplotlib.pyplot as plt
from datetime import datetime
from collections import defaultdict

nltk.download('punkt')
nltk.download('vader_lexicon')

#Initialize GNews client
google_news = GNews(
    language='en',
    country='US',
    period='7d',           
    max_results=100        
)

#Search for news using Gnews
articles = google_news.get_news('Apple stock')

dictionary = {
    'title': [],
    'url': [],
    'publisher': [],
    'published date': [],
    'description': [],
    'sentiment': []
}

#Prints titles and performs sentiment analysis on the descriptions
for idx, article in enumerate(articles):
    title = article['title']
    date = article['published date']
    url = article['url']
    publisher = article['publisher']
    description = article['description']

    #Sentiment analysis
    sia = SentimentIntensityAnalyzer()
    sentiment = sia.polarity_scores(description)

    #Save to a dictionary
    dictionary['title'].append(title)
    dictionary['url'].append(url)
    dictionary['publisher'].append(publisher)
    dictionary['published date'].append(date)
    dictionary['description'].append(description)
    dictionary['sentiment'].append(sentiment)

    print(f"\nArticle #{idx+1} Title: {title}")
    print(f"Published Date: {date}")
    print("Sentiment:", sentiment)
    print(f'description: {description[:100]}...')  # Print first 100 chars of description

#Aggregates by date
sentiment_by_date = defaultdict(list)
for date, sentiment in zip(dictionary['published date'], dictionary['sentiment']):
    # Parse date string to date object (ignore time)
    try:
        date_obj = datetime.strptime(date[:16], '%a, %d %b %Y')
        sentiment_by_date[date_obj.date()].append(sentiment['compound'])
    except Exception as e:
        continue

#Compute average sentiment per day
avg_sentiment = {d: sum(vals)/len(vals) for d, vals in sentiment_by_date.items() if vals}

#sort by date
sorted_dates = sorted(avg_sentiment.keys())
sorted_sentiments = [avg_sentiment[d] for d in sorted_dates]

plt.figure(figsize=(10, 5))
plt.plot(sorted_dates, sorted_sentiments, marker='o')
plt.title('Average Sentiment per Day for Apple from Google News descriptions')
plt.xlabel('Date')
plt.ylabel('Average Compounded Sentiment')
plt.grid(True)
plt.tight_layout()
plt.show()

Student GPA Predictor

Neural Network GPA Predictor
A machine learning project that predicts student GPA based on lifestyle factors using a neural network built with TensorFlow/Keras. The model uses features such as study hours, extracurricular activities, sleep, social, and physical activity hours to estimate GPA.

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense
from tensorflow.keras.optimizers import Adam

# Load dataset
df = pd.read_csv('student_lifestyle_dataset.csv')

# Prepare features and target
X = df[['Study_Hours_Per_Day', 'Extracurricular_Hours_Per_Day', 
        'Sleep_Hours_Per_Day', 'Social_Hours_Per_Day', 
        'Physical_Activity_Hours_Per_Day']].values
y = df['GPA'].values

# Split and scale
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# Build and train model
model = Sequential([
    Dense(5, activation='relu', input_shape=(X_train.shape[1],)),
    Dense(64, activation='relu'),
    Dense(128, activation='relu'),
    Dense(64, activation='relu'),
    Dense(1)
])
model.compile(optimizer=Adam(learning_rate=0.0001), loss='mean_squared_error', metrics=['mae'])
history = model.fit(X_train, y_train, validation_data=(X_test, y_test), epochs=25, batch_size=8, verbose=1)

# Evaluate and plot
test_loss, test_mae = model.evaluate(X_test, y_test, verbose=1)
print(f"Test Mean Absolute Error (MAE): {test_mae:.2f}")
y_pred = model.predict(X_test)
plt.figure(figsize=(8, 6))
plt.scatter(y_test, y_pred, color='blue', label='Predictions')
plt.plot([min(y_test), max(y_test)], [min(y_test), max(y_test)], 'r--', label='Perfect Fit')
plt.xlabel('actual GPA')
plt.ylabel('predicted GPA')
plt.legend()
plt.title('Neural Network Predictions vs Actual GPA')
plt.show()

Store Analysis

Store Analysis: Correlation and Customer Insights
This analysis explores the relationship between annual income, time spent on the website, and purchasing behavior. Using Python, pandas, seaborn, and matplotlib, we visualize correlations and trends in customer data.

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

df = pd.read_excel('Project (customer_purchase_data).xlsx')

# Correlation heatmap
corr = df.corr(numeric_only=True)
plt.figure(figsize=(10, 8))
sns.heatmap(corr, annot=True, cmap='coolwarm', fmt='.2f')
plt.title('Correlation Heatmap of All Variables')
plt.show()

# Annual income vs. time spent
plt.figure(figsize=(8, 6))
sns.barplot(data=df, x='AnnualIncome', y='TimeSpentOnWebsite')
plt.title('Annual Income vs. Time Spent on Website')
plt.xlabel('Annual Income (k$)')
plt.ylabel('Time Spent on Website (minutes)')
plt.show()

# Binned income groups
income_bins = np.linspace(df['AnnualIncome'].min(), df['AnnualIncome'].max(), 10)
df['IncomeGroup'] = pd.cut(df['AnnualIncome'], bins=income_bins)
mean_time_by_income = df.groupby('IncomeGroup')['TimeSpentOnWebsite'].mean().reset_index()
plt.figure(figsize=(10, 6))
sns.barplot(data=mean_time_by_income, x='IncomeGroup', y='TimeSpentOnWebsite', color='skyblue')
plt.xticks(rotation=45)
plt.title('Average Time Spent on Website by Annual Income Group')
plt.xlabel('Annual Income Group')
plt.ylabel('Average Time Spent on Website (minutes)')
plt.tight_layout()
plt.show()

The analysis shows no significant correlation between annual income and time spent on the website.

By grouping users by age and calculating the average number of purchases for each age group, the results highlighted that younger customers tend to make more purchases. This insight can help inform targeted marketing strategies and customer engagement efforts.

Made Up SQL Sales Project

Project Overview & Explanations
  • Data Generation: Python and Faker were used to generate thousands of fake sales records, simulating realistic business data for analysis.
  • SQL Analysis: A variety of SQL queries were written to explore the data, including totals, averages, groupings, subqueries, and time-based summaries. These queries provide insights into sales performance, customer behavior, and product trends.
  • Visualization: The sales_over_time.py script demonstrates how to extract and visualize sales trends using Python and matplotlib, making it easy to spot patterns and communicate results.
  • Results Display: The query results are shown below as HTML tables, making it easy to review the output of each analysis step.

Python: Fill Sales Table
import sqlite3
from faker import Faker
import random

conn = sqlite3.connect('database.db')
cursor = conn.cursor()
fake = Faker()
products = ['Laptop', 'Phone', 'Tablet', 'Monitor', 'Keyboard', 'Mouse', 'Printer', 'Camera']
rows = []
for _ in range(5000):
    date = fake.date_between(start_date='-2y', end_date='today')
    customer_name = fake.name()
    product = random.choice(products)
    quantity = random.randint(1, 10)
    price = round(random.uniform(50, 2000), 2)
    total = round(quantity * price, 2)
    rows.append((str(date), customer_name, product, quantity, price, total))
cursor.executemany('INSERT INTO sales (date, customer_name, product, quantity, price, total) VALUES (?, ?, ?, ?, ?, ?)', rows)
conn.commit()
conn.close()
All SQL Queries Used
-- Show the first 10 sales records
SELECT * FROM sales LIMIT 10;

-- Count the total number of sales records
SELECT COUNT(*) AS total_sales FROM sales;

-- Get total sales per product
SELECT product, SUM(total) AS total_revenue FROM sales GROUP BY product ORDER BY total_revenue DESC;

-- Find the top 5 customers by total spending
SELECT customer_name, SUM(total) AS total_spent FROM sales GROUP BY customer_name ORDER BY total_spent DESC LIMIT 5;

-- Get sales summary by month
SELECT strftime('%Y-%m', date) AS month, SUM(total) AS monthly_sales FROM sales GROUP BY month ORDER BY month DESC;

-- Calculate the average sale value per product
SELECT product, AVG(total) AS avg_sale_value FROM sales GROUP BY product ORDER BY avg_sale_value DESC;

-- Find the day with the highest total sales
SELECT date, SUM(total) AS daily_total FROM sales GROUP BY date ORDER BY daily_total DESC LIMIT 1;

-- List products that have sold more than 100 units in total
SELECT product, SUM(quantity) AS total_units_sold FROM sales GROUP BY product HAVING total_units_sold > 100 ORDER BY total_units_sold DESC;

-- Subquery: Find customers who have made more than 5 purchases
SELECT customer_name, COUNT(*) AS num_purchases FROM sales GROUP BY customer_name HAVING num_purchases > 5 ORDER BY num_purchases DESC;

-- Subquery: List sales where the total is above the average sale total
SELECT * FROM sales WHERE total > (SELECT AVG(total) FROM sales) ORDER BY total DESC LIMIT 10;

-- Subquery: For each product, show the highest single sale (by total)
SELECT product, MAX(total) AS max_single_sale FROM sales GROUP BY product ORDER BY max_single_sale DESC;

-- Subquery: Find the top 3 products by total revenue in the last 6 months
SELECT product, SUM(total) AS recent_revenue FROM sales WHERE date >= date('now', '-6 months') GROUP BY product ORDER BY recent_revenue DESC LIMIT 3;

-- Quick sales figure: Total sales for the current year
SELECT SUM(total) AS total_sales_this_year FROM sales WHERE strftime('%Y', date) = strftime('%Y', 'now');

These SQL queries were used to analyze and summarize the sales data, including aggregations, subqueries, and time-based calculations.

Python: Monthly Sales Over Time Visualization
import sqlite3
import pandas as pd
import matplotlib.pyplot as plt

conn = sqlite3.connect('database.db')

query = """
SELECT strftime('%Y-%m', date) AS month, SUM(total) AS monthly_sales
FROM sales
GROUP BY month
ORDER BY month;
"""
df = pd.read_sql_query(query, conn)
conn.close()

plt.figure(figsize=(10, 6))
plt.plot(df['month'], df['monthly_sales'], marker='o')
plt.title('Monthly Sales Over Time')
plt.xlabel('Month')
plt.ylabel('Total Sales')
plt.xticks(rotation=45)
plt.tight_layout()
plt.grid(True)
plt.show()
